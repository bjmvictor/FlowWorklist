# Install RECEIVER as service
nssm install PRINTERSCU-Receiver "C:\dcmtk\bin\dcmprscp.exe" -c "C:\dicom-print\printer.cfg" -p PHILIPS_GATEWAY +d -v
nssm set PRINTSCU-Receiver AppDirectory "C:\dicom-print"
nssm set PRINTSCU-Receiver AppEnvironmentExtra "PATH=C:\dcmtk\bin;%PATH%"
nssm set PRINTSCU-Receiver AppStdout "C:\dicom-print\logs\receiver.out.log"
nssm set PRINTSCU-Receiver AppStderr "C:\dicom-print\logs\receiver.err.log"
nssm set PRINTSCU-Receiver AppRotateFiles 1
nssm set PRINTSCU-Receiver AppRotateOnline 1
nssm set PRINTSCU-Receiver AppRotateSeconds 86400
nssm set PRINTSCU-Receiver AppRotateBytes 10485760
nssm set PRINTSCU-Receiver Start SERVICE_AUTO_START
nssm set PRINTSCU-Receiver AppExit Default Restart
nssm set PRINTSCU-Receiver AppRestartDelay 3000
